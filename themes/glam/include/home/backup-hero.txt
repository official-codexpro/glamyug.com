<script>
    gsap.registerPlugin(ScrollTrigger);

    const lenis = new Lenis();
    function raf(time) { lenis.raf(time); requestAnimationFrame(raf); }
    requestAnimationFrame(raf);

    const wrapper = document.getElementById('sliderWrapper');
    const slides = document.querySelectorAll('.slide');
    const totalRealSlides = slides.length - 1; 
    const stepPercent = 100 / slides.length; 
    
    let currentIndex = 0;
    let isTransitioning = false;
    const transitionDuration = 1.4;

    function moveSlider(direction) {
        if (isTransitioning) return;
        isTransitioning = true;
        
        currentIndex += direction;
        slides.forEach(s => s.classList.remove('active'));

        if (currentIndex > totalRealSlides) {
             gsap.to(wrapper, {
                xPercent: -(totalRealSlides * stepPercent),
                duration: transitionDuration,
                ease: "expo.inOut",
                onComplete: () => {
                    currentIndex = 0;
                    gsap.set(wrapper, { xPercent: 0 });
                    slides[currentIndex].classList.add('active');
                    isTransitioning = false;
                }
            });
        } 
        else if (currentIndex < 0) {
            gsap.set(wrapper, { xPercent: -(totalRealSlides * stepPercent) }); 
            currentIndex = totalRealSlides - 1;
            
            gsap.to(wrapper, {
                xPercent: -(currentIndex * stepPercent),
                duration: transitionDuration,
                ease: "expo.inOut",
                onComplete: () => {
                    slides[currentIndex].classList.add('active');
                    isTransitioning = false;
                }
            });
        }
        else {
            gsap.to(wrapper, {
                xPercent: -(currentIndex * stepPercent),
                duration: transitionDuration,
                ease: "expo.inOut",
                onComplete: () => {
                    slides[currentIndex].classList.add('active');
                    isTransitioning = false;
                }
            });
        }
    }

    let autoPlay = setInterval(() => moveSlider(1), 5000);

    function manualMove(dir) {
        if (isTransitioning) return;
        clearInterval(autoPlay);
        moveSlider(dir);
        autoPlay = setInterval(() => moveSlider(1), 5000);
    }

    const tl = gsap.timeline({
        scrollTrigger: {
            trigger: "#mainTrigger",
            start: "top top",
            end: "bottom bottom",
            scrub: 1.2,
            onUpdate: (self) => {
                const controls = document.getElementById('sliderControls');
                controls.style.opacity = self.progress > 0.05 ? "0" : "1";
            }
        }
    });

    tl.to("#heroBanner", { yPercent: -100, ease: "power2.inOut" }, 0);
    tl.to("#revealContent", { opacity: 1, y: -40, ease: "power2.out" }, 0.3);
</script>
